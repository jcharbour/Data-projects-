Create conditional probability table fpr each node
cptW = matrix(c(0.6,0.4), ncol = 2, dimnames = list(NULL, c("true","false")))

cptS = c(0.2,0.8,0.75,0.25)
dim(cptS) = c(2,2)
dimnames(cptS) = list("S" = c("true","false"), "W" = c("true","false"))

cptR = c(0.8,0.2,0.1,0.9)
dim(cptR) = c(2,2)
dimnames(cptR) = list("R" = c("true","false"), "W" = c("true","false"))

cptSR = c(0.7,0.3,0,1)
dim(cptSR) = c(2,2)
dimnames(cptSR) = list("SR" = c("true","false"), "R" = c("true","false"))

cptWG = c(0.95,0.05,0.9,0.1,0.8,0.2,0,1)
dim(cptWG) = c(2,2,2)
dimnames(cptWG) = list("WG" = c("true","false"),"S" = c("true","false"),"R" = c("true","false"))
dfit = custom.fit(e, dist = list(W = cptW, S = cptS,
                                 WG = cptWG, R= cptR, 
                                 SR = cptSR))
